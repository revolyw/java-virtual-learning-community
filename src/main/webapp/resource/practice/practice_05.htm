//单机各按钮的处理操作

   public void actionPerformed(ActionEvent e)

   {

      //获取按键内容

      JButton but=(JButton)e.getSource();

 

      //是否输入数字

   if(but==b[0]||but==b[1]||but==b[2]||but==b[3]||but==b[4]||but==b[5]||but==b[6]||but==b[7]||but==b[8]||but==b[9]||but==b8)

         Sum(e.getActionCommand());

      //是否输入运算符

      else if(but==b1||but==b2||but==b3||but==b4||but==b6)

         Sign(e.getActionCommand());

      //进行取倒数操作

      else if(but==b5)

         Reciprocal(e.getActionCommand());

      //进行开方操作

      else if(but==b7)

         Sqrt(e.getActionCommand());

      //进行去相反数

      else if(but==b9)

         Opposite(e.getActionCommand());

      //进行运算操作

      else if(but==b0)

         Count(e.getActionCommand());

      //去除末尾数字

      else if(but==a0)

         Back(e.getActionCommand());

      //清空文本框

      else if(but==a1)

         Clear(e.getActionCommand());        

   }

 

   //按数字键触发

   public void Sum(String a)

   {

      str2=a;

      if(Flags==false)

      {

         str=str+a;

         tf.setText(str);

         sum=Double.parseDouble(str);

      }

      else if(Flags)

      {

         str1=str1+a;

         tf.setText(str1);

         sum1=Double.parseDouble(str1);

      }

   }

 

   //按+，-，*，/，%触发

   public void Sign(String a)

   {

      tf.setText(a);

      sign=a;

      Flags=true;

   }

 

   //按1/x键触发

   public void Reciprocal(String a)

   {

 

      if(!Flags)

      {

         sum=1/sum;

         tf.setText(""+sum);

      }

      else

      {

         sum1=1/sum1;

         tf.setText(""+sum1);

      }

   }

 

   //按sqrt触发

   public void Sqrt(String a)

   {

 

      if(!Flags)

      {

         sum=Math.sqrt(sum);

         tf.setText(""+sum);

      }

      else

      {

         sum1=Math.sqrt(sum1);

         tf.setText(""+sum1);

      }

   }

 

   //按+/-触发

   public void Opposite(String a)

   {

      if(!Flags)

      {

         sum=-sum;

         tf.setText(""+sum);

      }

      else

      {

         sum1=-sum1;

         tf.setText(""+sum1);

      }

   }

 

   //按=触发

   public void Count(String a)

   {

      if(sign.equals("+"))

      {

         sum=sum+sum1;

         tf.setText(""+sum);

      }

      else if(sign.equals("-"))

      {

         sum=sum-sum1;

         tf.setText(""+sum);

      }

      else if(sign.equals("×"))

      {

         sum=sum*sum1;

         tf.setText(""+sum);

      }

      else if(sign.equals("/"))

         if(sum1==0)

            tf.setText("除数不可为0！");

         else

         {

            sum=sum/sum1;

            tf.setText(""+sum);

         }

      else if(sign.equals("%"))

      {

         sum=sum%sum1;

         tf.setText(""+sum);

      }

      //sum=0;

      sum1=0;

      str="\0";

      str1="\0";

      Flags=false;

   }

 

   //按Back 触发

   public void Back(String a)

   {    

      if(!Flags)

      {

 

         tf.setText(str.substring(0, str.length()-1));

         str=tf.getText();

         sum=Double.parseDouble(str); 

      }

      else

      {

         tf.setText(str1.substring(0, str1.length()-1));

         str1=tf.getText();

         sum1=Double.parseDouble(str1); 

      }

   }

 

   //按C触发

   public void Clear(String a)

   {

      str="\0";

      str1="\0";

      sum=0;

      sum1=0;

      Flags=false;

      tf.setText(str);

   }
