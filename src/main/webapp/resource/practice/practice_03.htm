Java小程序——九宫格

九宫格是一种传统的游戏程序，通过生成随机排列的数字方块，允许用户童工单机来移动方块，不过只有靠近空格的方块才能被移动。当所有的数字按序排列整齐后，游戏程序即告结束。本范例演示了一种简单的九宫格程序，重点请注意移动方块的实现方法。

一．创建基本窗体

该阶段主要定义了九宫格窗体类DemoWindow,该类包含一个显示九宫格界面的面板类GridPanel和一个控制面板类ControlPanel。GridPanel类的构造函数接收一个整型参数，可以生成指定行列数的随机数组和相应的按钮数组，两者一一对应，按钮的标签是对应数值数组元素的数值，如果数值为最大值，则对应按钮显示空格标签。

   privatevoid initializeButton(JButton[] jbs) {

      // 循环遍历每个数值数组元素

      for (int i = 0; i < jbs.length; i++) {

         if (records[i] != number * number)

            jbs[i].setText(String.valueOf(records[i]));

         // 如果是最大的数，将对应的按钮标题设置为空

         else

            jbs[i].setText("");

      }

   }

二．添加单击鼠标事件

在用户单击按钮时，程序判断当前按钮的方位和与空格按钮的位置关系，并适时交换按钮标签和相应的数值数组元素。在每一次移动操作后，程序都会判断数值数组是否已经排好序列，即判断是否移动成功。ControlPanel类提供了重来游戏和退出游戏的控制功能。

// 响应单击按钮

   publicvoid actionPerformed(ActionEvent e) {

      JButton jb = (JButton) e.getSource();

 

      // 不处理单击空白按钮

      if (!jb.getText().equals("")) {

         // 获取当前按钮的标题

         int currentNumber = Integer.parseInt(jb.getText());

 

         // 获取对应数字在数值数组中的位置

         int currentPos = -1;

         for (int i = 0; i <records.length; i++) {

            if (records[i] == currentNumber) {

               currentPos = i;

               break;

            }

         }

 

         // 如果单击的按钮不位于第一行，并且上一行的对应按钮标题为空，交换按钮标题和对应的数值数组元素

         if (currentPos >= number

               &&records[currentPos - number] == number * number) {

            exchangetTwoNumber(records, currentPos, currentPos - number);

            exchangeLabel(currentPos, currentPos - number);

         }

 

         // 如果单击的按钮不位于最后一行，并且下一行的对应按钮标题为空，交换按钮标题和对应的数值数组元素

         if (currentPos <number * number - number

               &&records[currentPos + number] == number * number) {

            exchangetTwoNumber(records, currentPos, currentPos + number);

            exchangeLabel(currentPos, currentPos + number);

         }

 

         // 如果单击的按钮不位于第一列，并且左列的对应按钮标题为空，交换按钮标题和对应的数值数组元素

         if (currentPos % number != 0

               &&records[currentPos - 1] == number * number) {

            exchangetTwoNumber(records, currentPos, currentPos - 1);

            exchangeLabel(currentPos, currentPos - 1);

         }

 

         // 如果单击的按钮不位于最后一列，并且右列的对应按钮标题为空，交换按钮标题和对应的数值数组元素

         if (currentPos % number != number - 1

               &&records[currentPos + 1] == number * number) {

            exchangetTwoNumber(records, currentPos, currentPos + 1);

            exchangeLabel(currentPos, currentPos + 1);

         }

 

         // 如果数值数组移动成功，退出循环

         boolean flag = false;

         for (int i = 0; i <records.length; i++) {

            if (records[i] != i + 1) {

               flag = true;

               break;

            }

         }

 

         // 如果移动成功，显示消息对话框

         if (flag == false)

            JOptionPane.showMessageDialog(null, "成功！");

      }

   }

 

   // 该函数初始化用于检查是否移动成功的数值数组

   privatevoid initializeRandomNumber(int[] records) {

      // 生成0到9十个数值

      for (int i = 0; i < records.length; i++) {

         records[i] = i + 1;

      }

 

      // 随机选取两个数值数值元素，交换数值，总次数为100次，产生混淆次序的效果

      for (int i = 0; i < 100; i++) {

         int beginPos = (int) (Math.random() * number * number);

         int endPos = (int) (Math.random() * number * number);

 

         // 交换数值数组的两个元素值

         exchangetTwoNumber(records, beginPos, endPos);

      }

   }
